async function init(regionCode) {

    CONFIG.userRegion = regionCode;

   

    // Fetch real-time status from your Google Apps Script API

    const response = await fetch('YOUR_APPS_SCRIPT_WEB_APP_URL?action=getRegion&region=' + regionCode);

    const data = await response.json();

   

    document.getElementById('node-location').innerText = `NODE_${regionCode}`;

   

    // The "Merit-Lock" logic we built

    if (data.isMeritZone) {

        document.getElementById('donation-gate').classList.add('hidden');

    }

}
